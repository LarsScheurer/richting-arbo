import React, { useState, useEffect } from 'react';
import CustomerDetail from './CustomerDetail';
import { customerService } from './services/CustomerService'; // Zie bestand 3
// Importeer hier je iconen of gebruik placeholders

const CustomerOverview = ({ user, onOpenDoc }) => {
    const [customers, setCustomers] = useState([]);
    const [allUsers, setAllUsers] = useState([]);
    const [selectedCustomer, setSelectedCustomer] = useState(null);
    const [selectedIds, setSelectedIds] = useState([]);
    const [showAddModal, setShowAddModal] = useState(false);
    const [loading, setLoading] = useState(true);

    // Form state voor nieuwe klant
    const [newName, setNewName] = useState("");
    const [newIndustry, setNewIndustry] = useState("");
    const [newWebsite, setNewWebsite] = useState("");
    const [newLogoUrl, setNewLogoUrl] = useState("");
    const [assignedUserIds, setAssignedUserIds] = useState([user.id]);

    useEffect(() => {
        const loadData = async () => {
            setLoading(true);
            const fetchedCustomers = await customerService.getCustomersForUser(user.id, user.role); //
            const fetchedUsers = await customerService.getAllUsers(); //
            setCustomers(fetchedCustomers);
            setAllUsers(fetchedUsers);
            setLoading(false);
        };
        loadData();
    }, [user]);

    const handleCreateCustomer = async () => {
        if (!newName) return;
        
        const newCustomer = {
            id: `cust_${Date.now()}`,
            name: newName,
            industry: newIndustry,
            website: newWebsite,
            logoUrl: newLogoUrl,
            status: 'active',
            assignedUserIds: assignedUserIds,
            createdAt: new Date().toISOString()
        };

        await customerService.addCustomer(newCustomer); //
        setCustomers(prev => [...prev, newCustomer]);
        setShowAddModal(false);
        
        // Reset form
        setNewName("");
        setNewIndustry("");
        setNewWebsite("");
    };

    const toggleUserAssignment = (userId) => {
        setAssignedUserIds(prev => 
            prev.includes(userId) ? prev.filter(id => id !== userId) : [...prev, userId]
        );
    };

    // Als er een klant geselecteerd is, toon het detailscherm
    if (selectedCustomer) {
        return (
            <CustomerDetail 
                customer={selectedCustomer} 
                user={user}
                onBack={() => setSelectedCustomer(null)}
                onUpdate={(updated) => {
                    setCustomers(prev => prev.map(c => c.id === updated.id ? updated : c));
                    setSelectedCustomer(updated);
                }}
                onDelete={(id) => {
                    setCustomers(prev => prev.filter(c => c.id !== id));
                    setSelectedCustomer(null);
                }}
                onOpenDoc={onOpenDoc}
            />
        );
    }

    return (
        <div className="space-y-6">
            <div className="flex justify-between items-center">
                <h2 className="text-2xl font-bold text-slate-900">Klanten & Dossiers</h2>
                <button 
                    onClick={() => setShowAddModal(true)}
                    className="bg-richting-orange text-white px-4 py-2 rounded-lg font-bold hover:bg-orange-600 transition-colors flex items-center gap-2"
                >
                    <span className="text-lg">+</span> Nieuwe Klant
                </button>
            </div>

            {loading ? (
                <div className="text-center py-10 text-gray-500">Laden...</div>
            ) : (
                <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
                    {customers.map(customer => (
                        <div 
                            key={customer.id}
                            onClick={() => setSelectedCustomer(customer)}
                            className="bg-white rounded-xl shadow-sm border border-gray-200 p-6 hover:shadow-md cursor-pointer transition-all group"
                        >
                            <div className="flex justify-between items-start mb-4">
                                {/* Logo placeholder */}
                                <div className="w-12 h-12 rounded-lg bg-gray-50 border border-gray-100 flex items-center justify-center overflow-hidden">
                                    {customer.logoUrl ? (
                                        <img src={customer.logoUrl} alt={customer.name} className="w-10 h-10 object-contain" />
                                    ) : (
                                        <span className="text-xl">üè¢</span>
                                    )}
                                </div>
                                <span className={`px-2 py-1 rounded-full text-xs font-bold uppercase ${customer.status === 'active' ? 'bg-green-100 text-green-700' : 'bg-gray-100 text-gray-600'}`}>
                                    {customer.status}
                                </span>
                            </div>
                            <h3 className="text-lg font-bold text-slate-900 mb-1 group-hover:text-richting-orange">
                                {customer.name}
                            </h3>
                            <p className="text-sm text-gray-500 mb-6">
                                {customer.industry || "Geen branche"}
                            </p>
                            <div className="border-t border-gray-100 pt-4 flex justify-between items-center">
                                <span className="text-xs text-gray-400">Bekijk Dossier ‚Üí</span>
                            </div>
                        </div>
                    ))}
                </div>
            )}

            {/* Modal voor toevoegen (vereenvoudigd voor leesbaarheid) */}
            {showAddModal && (
                <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/50 backdrop-blur-sm">
                    <div className="bg-white rounded-xl shadow-2xl w-full max-w-lg overflow-hidden">
                        <div className="bg-gray-50 px-6 py-4 border-b flex justify-between">
                            <h3 className="font-bold">Nieuwe Klant</h3>
                            <button onClick={() => setShowAddModal(false)}>‚úï</button>
                        </div>
                        <div className="p-6 space-y-4">
                            <input 
                                className="w-full border p-2 rounded" 
                                placeholder="Bedrijfsnaam"
                                value={newName} onChange={e => setNewName(e.target.value)} 
                            />
                            <input 
                                className="w-full border p-2 rounded" 
                                placeholder="Branche"
                                value={newIndustry} onChange={e => setNewIndustry(e.target.value)} 
                            />
                            {/* Team selectie sectie */}
                            <div>
                                <label className="text-sm font-bold mb-2 block">Team Toegang</label>
                                <div className="border rounded max-h-40 overflow-y-auto">
                                    {allUsers.map(u => (
                                        <div 
                                            key={u.id} 
                                            onClick={() => toggleUserAssignment(u.id)}
                                            className={`p-2 flex items-center gap-2 cursor-pointer ${assignedUserIds.includes(u.id) ? 'bg-orange-50' : ''}`}
                                        >
                                            <div className={`w-4 h-4 border rounded flex items-center justify-center ${assignedUserIds.includes(u.id) ? 'bg-orange-500 border-orange-500' : ''}`}>
                                                {assignedUserIds.includes(u.id) && <span className="text-white text-xs">‚úì</span>}
                                            </div>
                                            <span className="text-sm">{u.name}</span>
                                        </div>
                                    ))}
                                </div>
                            </div>
                            <button onClick={handleCreateCustomer} className="w-full bg-richting-orange text-white py-3 rounded font-bold">
                                Aanmaken
                            </button>
                        </div>
                    </div>
                </div>
            )}
        </div>
    );
};

export default CustomerOverview;
