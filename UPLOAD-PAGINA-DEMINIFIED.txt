import React, { useState, useEffect } from 'react';
import { analyzeDocument } from './services/GeminiService'; // Die we eerder maakten
import { documentService } from './services/DocumentService';
import { customerService } from './services/CustomerService';

const CONTENT_TYPES = {
    TEXT: 'TEXT',
    URL: 'URL',
    EMAIL: 'EMAIL',
    GOOGLE_DOC: 'GOOGLE_DOC',
    PDF: 'PDF'
};

const UploadPage = ({ user, onUploadComplete }) => {
    const [step, setStep] = useState(1);
    const [contentType, setContentType] = useState(CONTENT_TYPES.TEXT);
    
    // Form state
    const [title, setTitle] = useState("");
    const [content, setContent] = useState(""); [cite_start]// p in original [cite: 1776]
    const [url, setUrl] = useState(""); // T in original
    const [selectedCustomerId, setSelectedCustomerId] = useState(""); // x in original
    
    const [customers, setCustomers] = useState([]);
    const [isAnalyzing, setIsAnalyzing] = useState(false);
    const [aiResult, setAiResult] = useState(null); // he in original

    useEffect(() => {
        const loadCustomers = async () => {
            const custs = await customerService.getCustomersForUser(user.id, user.role);
            setCustomers(custs);
        };
        loadCustomers();
    }, [user]);

    const handleAnalyze = async () => {
        if (!title) {
            alert("Geef aub een titel op");
            return;
        }
        if (!content && !url) return;

        setIsAnalyzing(true);

        [cite_start]// Prompt voorbereiden op basis van input type [cite: 1777]
        let textToAnalyze = content;
        if (contentType === CONTENT_TYPES.URL || contentType === CONTENT_TYPES.GOOGLE_DOC || contentType === CONTENT_TYPES.PDF) {
            textToAnalyze = `URL: ${url}\n(Inhoud niet direct beschikbaar, genereer op basis van URL of Titel: ${title})`;
        } else if (contentType === CONTENT_TYPES.EMAIL) {
            textToAnalyze = `EMAIL ONDERWERP: ${title}\nINHOUD:\n${content}`;
        }

        const result = await analyzeDocument(textToAnalyze); // De functie uit GeminiService
        setAiResult(result);
        setIsAnalyzing(false);
        setStep(2);
    };

    const handleSave = async () => {
        if (!aiResult) return;

        const newDoc = {
            id: `doc_${Date.now()}`,
            title: title || "Nieuwe Bron",
            content: content || url,
            originalUrl: url || "",
            type: contentType,
            uploadedBy: user.id,
            uploadedAt: new Date().toISOString(),
            
            [cite_start]// AI Data [cite: 1778]
            summary: aiResult.summary,
            mainCategoryId: aiResult.mainCategoryId,
            subCategoryId: aiResult.subCategoryId,
            tags: aiResult.tags,
            
            viewedBy: [],
            likedBy: [],
            isArchived: false,
            customerId: selectedCustomerId || undefined
        };

        await documentService.addDocument(newDoc);
        onUploadComplete();
    };

    return (
        <div className="max-w-3xl mx-auto bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden">
            <div className="bg-gray-50 px-6 py-4 border-b flex justify-between items-center">
                <h3 className="font-bold text-slate-800">Nieuwe Kennis Toevoegen</h3>
                <div className="flex gap-2">
                    <span className={`w-3 h-3 rounded-full ${step >= 1 ? "bg-richting-orange" : "bg-gray-300"}`}></span>
                    <span className={`w-3 h-3 rounded-full ${step >= 2 ? "bg-richting-orange" : "bg-gray-300"}`}></span>
                </div>
            </div>

            <div className="p-8">
                {step === 1 && (
                    <div className="space-y-6">
                        {/* Type Selectie */}
                        <div>
                            <label className="block text-sm font-medium text-gray-700 mb-2">Type Bron</label>
                            <div className="grid grid-cols-3 gap-2">
                                {[
                                    { id: CONTENT_TYPES.TEXT, label: "Tekst", icon: "ðŸ“" },
                                    { id: CONTENT_TYPES.URL, label: "Website", icon: "ðŸ”—" },
                                    { id: CONTENT_TYPES.EMAIL, label: "Email", icon: "ðŸ“§" },
                                    { id: CONTENT_TYPES.GOOGLE_DOC, label: "G-Doc", icon: "ðŸ“„" },
                                    { id: CONTENT_TYPES.PDF, label: "PDF", icon: "ðŸ“‘" },
                                ].map(type => (
                                    <button
                                        key={type.id}
                                        onClick={() => setContentType(type.id)}
                                        className={`py-3 px-2 rounded-lg border-2 text-xs font-bold flex flex-col items-center gap-1 ${contentType === type.id ? "border-richting-orange text-richting-orange bg-orange-50" : "border-gray-200 text-gray-500 hover:border-gray-300"}`}
                                    >
                                        <span>{type.icon}</span> {type.label}
                                    </button>
                                ))}
                            </div>
                        </div>

                        {/* Inputs */}
                        <input
                            type="text"
                            className="w-full px-3 py-2 border border-gray-300 rounded-md focus:ring-richting-orange"
                            placeholder={contentType === CONTENT_TYPES.EMAIL ? "Onderwerp" : "Titel"}
                            value={title}
                            onChange={e => setTitle(e.target.value)}
                        />

                        <select
                            className="w-full px-3 py-2 border border-gray-300 rounded-md bg-white"
                            value={selectedCustomerId}
                            onChange={e => setSelectedCustomerId(e.target.value)}
                        >
                            <option value="">-- Koppel aan Klant (Optioneel) --</option>
                            {customers.map(c => <option key={c.id} value={c.id}>{c.name}</option>)}
                        </select>

                        {(contentType === CONTENT_TYPES.TEXT || contentType === CONTENT_TYPES.EMAIL) ? (
                            <textarea
                                className="w-full h-40 px-3 py-2 border border-gray-300 rounded-md focus:ring-richting-orange"
                                placeholder="Plak hier de inhoud..."
                                value={content}
                                onChange={e => setContent(e.target.value)}
                            />
                        ) : (
                            <input
                                type="url"
                                className="w-full px-3 py-2 border border-gray-300 rounded-md focus:ring-richting-orange"
                                placeholder="https://..."
                                value={url}
                                onChange={e => setUrl(e.target.value)}
                            />
                        )}

                        <button
                            onClick={handleAnalyze}
                            disabled={isAnalyzing || !title}
                            className="w-full bg-slate-900 text-white py-3 rounded-lg font-bold hover:bg-slate-800 transition-colors disabled:opacity-50"
                        >
                            {isAnalyzing ? "AI is aan het denken..." : "âœ¨ Analyseer & Categoriseer"}
                        </button>
                    </div>
                )}

                {step === 2 && aiResult && (
                    <div className="space-y-6">
                        <div className="bg-orange-50 border border-orange-100 p-4 rounded-lg">
                            <h4 className="font-bold text-richting-orange mb-2">âœ¨ AI Analyse Resultaat</h4>
                            <div className="space-y-3">
                                <div>
                                    <label className="text-xs text-gray-500 uppercase font-bold">Samenvatting</label>
                                    <p className="text-sm text-gray-800">{aiResult.summary}</p>
                                </div>
                                <div className="grid grid-cols-2 gap-4">
                                    <div>
                                        <label className="text-xs text-gray-500 uppercase font-bold">Categorie</label>
                                        <p className="text-sm font-medium">{aiResult.mainCategoryId}</p>
                                    </div>
                                    <div>
                                        <label className="text-xs text-gray-500 uppercase font-bold">Subcategorie</label>
                                        <p className="text-sm font-medium">{aiResult.subCategoryId}</p>
                                    </div>
                                </div>
                                <div>
                                    <label className="text-xs text-gray-500 uppercase font-bold">Tags</label>
                                    <div className="flex flex-wrap gap-2 mt-1">
                                        {aiResult.tags.map(tag => (
                                            <span key={tag} className="px-2 py-1 bg-white border rounded text-xs text-gray-600">#{tag}</span>
                                        ))}
                                    </div>
                                </div>
                            </div>
                        </div>

                        <div className="flex gap-4">
                            <button onClick={() => setStep(1)} className="flex-1 px-4 py-2 border border-gray-300 rounded-lg text-gray-600">Terug</button>
                            <button onClick={handleSave} className="flex-1 px-4 py-2 bg-richting-orange text-white rounded-lg font-bold">Opslaan</button>
                        </div>
                    </div>
                )}
            </div>
        </div>
    );
};

export default UploadPage;
